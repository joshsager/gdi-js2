Let's take it for a spin - main.js

In your JavaScript File


function loadApplication(){
  console.log("I have loaded");
  var template = document.getElementById("albumstpl").innerHTML;
  Mustache.parse(template);   // optional, speeds up future uses
  var rendered = Mustache.to_html(template, {name:"Joshua"});
  document.getElementById("albums").innerHTML = rendered;
}

Let's take it for a spin - index.html

In your JavaScript File


<body onload="loadApplication()">
  <section id="albums">
    ... loading
  </section>

  <script id="albumstpl" type="text/template">
      <h1>{{name}}</h1>
  </script>
</body>
